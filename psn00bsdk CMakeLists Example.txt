cmake_minimum_required(VERSION 3.21)

project(my_game LANGUAGES C ASM)

psn00bsdk_add_executable(my_game GPREL main.c)

# Link precompiled exception handler objects
target_link_libraries(my_game PRIVATE
    ${CMAKE_CURRENT_SOURCE_DIR}/lib/ExceptionHandler.o
    ${CMAKE_CURRENT_SOURCE_DIR}/lib/exception_handler.o
    ${CMAKE_CURRENT_SOURCE_DIR}/lib/hook.o
    ${CMAKE_CURRENT_SOURCE_DIR}/lib/data.o
    ${CMAKE_CURRENT_SOURCE_DIR}/lib/sio_low.o
)

target_include_directories(my_game PRIVATE include)
